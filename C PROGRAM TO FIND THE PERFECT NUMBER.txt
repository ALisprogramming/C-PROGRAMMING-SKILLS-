#include <stdio.h>
int main() {
    int start, end;
    printf("Enter range: ");
    scanf("%d %d", &start, &end);

    for (int i = start; i <= end; i++) {
        int sum = 0;
        for (int j = 1; j < i; j++) {
            if (i % j == 0) {
                sum += j;
            }
        }

        if (sum == i && i > 0) {
            printf("%d\n", i);
        }
    }
    return 0;
}
