#include <stdio.h>

int main() 
{
    int n, i;
    
    printf("Enter the value of n: ");
    scanf("%d", &n);
    
    printf("Squares of numbers from 1 to %d:\n", n);
    
    for (i = 1; i <= n; i = i + 1) 
    {
        int square = i * i;
        printf("%d squared is %d\n", i, square);
    }
    
    return 0;
}
