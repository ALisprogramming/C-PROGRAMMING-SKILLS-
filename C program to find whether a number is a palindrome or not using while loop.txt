#include <stdio.h>

int main() {
    int n, rev = 0, copy;
    
    printf("Enter a number (n): ");
    scanf("%d", &n);
    
    copy = n;
    
    while (n > 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }
    
    if (copy == rev) {
        printf("%d is a palindrome.\n", copy);
    } else {
        printf("%d is not a palindrome.\n", copy);
    }
    
    return 0;
}
