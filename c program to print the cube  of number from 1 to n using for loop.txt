#include <stdio.h>

int main() 
{
    int n, i;

    printf("Enter the value of n: ");
    scanf("%d", &n);

    printf("Cube of numbers from 1 to %d: \n", n);

    for (i = 1; i <= n; i = i + 1) 
    {
        int cube = i * i * i;
        printf("%d cube is %d\n", i, cube);
    }

    return 0;
}
